### 2. **Qualidades Técnicas do Frontend**

#### ✨ **O Que Foi Bem Feito**

- **Responsividade:** Bootstrap 5 mobile-first, todos os componentes adaptáveis
- **Acessibilidade:** Modo de alto contraste bem implementado (84px mínimo em botões, cores contrastantes)
- **PWA Ready:** `manifest.json` corretamente estruturado, pronto para Service Worker

1. **JavaScript Vazio:**
   - `app.js` contém apenas comentário
   - Lógica de alto contraste está inline nas páginas PHP
   - Sem Service Worker implementado (critical para PWA offline)

2. **Dados Mockados:**
   - Todas as ofertas, KPIs e métricas são hardcoded
   - Gráficos Chart.js usam dados estáticos
   - Sem persistência real (não há backend conectado)

3. **Formulários Sem Backend:**
   - `cadastrar_oferta.php` não faz validação ou envio
   - Sem integração com banco de dados
   - Sem armazenamento em LocalStorage (offline)

4. **Navegação Incompleta:**
   - Links para "Minhas Ofertas", "Buscar Ofertas", "Minhas Reservas" apontam para `#`
   - Não há fluxo real de filtragem ou busca

### **Fase 2: Frontend Integrado (2 semanas)**

#### 2.1 Sistema de Autenticação
```javascript
// autenticacao.js - Exemplo de implementação
fetch('/api/usuarios/login.php', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ email, senha })
})
.then(res => res.json())
.then(data => {
  localStorage.setItem('token_jwt', data.token);
  localStorage.setItem('usuario_tipo', data.usuario.tipo);
  window.location.href = data.usuario.tipo === 'feirante' 
    ? 'feirante.php' 
    : 'consumidor.php';
})
```

#### 2.2 Persistência em LocalStorage + Sincronização
```javascript
// offline-sync.js - Enfileirar operações offline
const enfileirarOperacao = (tipo, dados) => {
  const fila = JSON.parse(localStorage.getItem('fila_sync') || '[]');
  fila.push({
    id: UUID.generate(),
    tipo,        // 'criar_oferta', 'reservar_kit'
    dados,
    timestamp: Date.now(),
    status: 'pendente'
  });
  localStorage.setItem('fila_sync', JSON.stringify(fila));
};

// Sincronizar ao reconectar
window.addEventListener('online', sincronizarFila);
```

#### 2.3 Unir JavaScript do app.js
```javascript
// assets/js/app.js - Consolidar lógica
import { autenticacao } from './modulos/autenticacao.js';
import { offlineSync } from './modulos/offline-sync.js';
import { ui } from './modulos/ui.js';

// Inicializar app
document.addEventListener('DOMContentLoaded', () => {
  autenticacao.verificarToken();
  offlineSync.iniciar();
  ui.inicializarModoAltoContraste();
  
  if ('serviceWorker' in navigator) {
    navigator.serviceWorker.register('sw.js');
  }
});
```

#### 2.4 Integração com Formulários
```php
<!-- cadastrar_oferta.php modificado -->
<form id="formOferta">
  <!-- campos existentes -->
</form>

<script>
document.getElementById('formOferta').addEventListener('submit', async (e) => {
  e.preventDefault();
  
  const formData = new FormData(e.target);
  const dados = Object.fromEntries(formData);
  
  try {
    const response = await fetch('/api/ofertas/criar.php', {
      method: 'POST',
      headers: { 'Authorization': `Bearer ${localStorage.getItem('token_jwt')}` },
      body: JSON.stringify(dados)

    if (resultado.sucesso) {
      mostrarMensagem('Oferta publicada!', 'sucesso');
    }
  } catch (erro) {
    // Offline: enfileirar
    enfileirarOperacao('criar_oferta', dados);
    mostrarMensagem('Será sincronizado quando reconectar', 'info');
  }
});
</script>
```

#### 4.2 **Problemas Identificados em UX/UI**

##### 🔴 **Problema A: Falta de Feedback Visual**

Cenário: Usuário clica em "Publicar Anúncio" em `cadastrar_oferta.php`

**Atual:**
- Nada acontece (sem backend)
- Sem spinner/loading
- Sem mensagem de sucesso/erro

**Impacto:** Usuário não sabe se ação foi processada

**Solução:**
```html
<button type="submit" class="btn btn-primary btn-lg" id="btnPublicar">
    <span id="btnText">Publicar Anúncio</span>
    <span id="spinner" class="spinner-border spinner-border-sm ms-2 visually-hidden" role="status">
        <span class="visually-hidden">Enviando...</span>
    </span>
</button>

<div id="alertContainer"></div>

<script>
document.getElementById('formOferta').addEventListener('submit', async (e) => {
    e.preventDefault();
    const btn = document.getElementById('btnPublicar');
    const spinner = document.getElementById('spinner');
    
    spinner.classList.remove('visually-hidden');
    btn.disabled = true;
    
    // Simular envio
    setTimeout(() => {
        mostrarAlerta('✅ Oferta publicada com sucesso!', 'sucesso');
        spinner.classList.add('visually-hidden');
        btn.disabled = false;
    }, 1000);
});

function mostrarAlerta(mensagem, tipo) {
    const alertDiv = document.createElement('div');
    alertDiv.className = `alert alert-${tipo} alert-dismissible fade show`;
    alertDiv.innerHTML = `${mensagem} <button type="button" class="btn-close" data-bs-dismiss="alert"></button>`;
    document.getElementById('alertContainer').appendChild(alertDiv);
}
</script>
```

formulario.addEventListener('submit', (e) => {
    if (!formulario.checkValidity()) {
        e.preventDefault();
        e.stopPropagation();
        validarCampo('produtoPreco', '#precoError');
    }
    formulario.classList.add('was-validated');
});

function validarCampo(id, seletorError) {
    const campo = document.getElementById(id);
    if (campo.value < 0.01) {
        document.querySelector(seletorError).textContent = 'Preço deve ser maior que R$ 0,00';
        campo.classList.add('is-invalid');
    } else {
        campo.classList.remove('is-invalid');
        campo.classList.add('is-valid');
    }
}

### **2. Session Storage vs LocalStorage**

| Caso de Uso | Recomendação | Motivo |
|---|---|---|
| Token JWT | SessionStorage | Limpa ao fechar browser (mais seguro) |
| Fila de sincronização | LocalStorage | Persiste entre sessions (crítico offline) |
| Preferências UI (alto contraste) | LocalStorage | User preference |
| Dados sensíveis temporários | SessionStorage | Menor superfície de ataque |

### **3. Tratamento de Erros em Offline/Online**

```javascript
// offline-handler.js
class OfflineHandler {
  static tratarErroOperacao(tipo, dados) {
    if (!navigator.onLine) {
      // Offline: enfileirar
      this.enfileirar(tipo, dados);
      return { sucesso: false, motivo: 'offline' };
    }
    
    // Online: retornar erro da rede
    return { sucesso: false, motivo: 'rede' };
  }
  
  static enfileirar(tipo, dados) {
    const fila = JSON.parse(localStorage.getItem('fila_sync') || '[]');
    fila.push({
      uuid: this.gerarUUID(),
      tipo,
      dados,
      timestamp: Date.now()
    });
    localStorage.setItem('fila_sync', JSON.stringify(fila));
  }
}
```

```php
<?php
class Database {
  private $host = getenv('DB_HOST') ?: 'localhost';
  private $db = getenv('DB_NAME') ?: 'xepaviva';
  private $user = getenv('DB_USER') ?: 'root';
  private $pass = getenv('DB_PASS') ?: '';
  
  private $conn;
  
  public function conectar() {
    try {
      $dsn = "mysql:host={$this->host};dbname={$this->db};charset=utf8mb4";
      $this->conn = new PDO($dsn, $this->user, $this->pass, [
        PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
        PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC
      ]);
      return $this->conn;
    } catch (PDOException $e) {
      die(json_encode(['erro' => 'Conexão falhou: ' . $e->getMessage()]));
    }
  }
}
```
